\begin{algorithm}[H]
\caption{跨通道全局特征池处理流程}
\label{alg:global_feature_pool}

\Input{
    $\mathcal{C} = \{ch_1, ch_2, \dots, ch_N\}$ — $N$ 个偏振通道状态（含局部ID、像素坐标、归一化坐标、速度）\;
    $\tau_{\text{dist}}$ — 空间关联距离阈值（取 $\text{MIN\_DIST}/2$）\;
    $\mathcal{P}$ — 上一帧全局特征绑定关系 $\{g_i \mapsto \{(ch, \text{lid})\}\}$\;
    $n_{\text{next}}$ — 下一个待分配的全局ID计数器\;
    $\mathcal{G}_{\text{grid}}$ — 上一帧空间哈希网格
}
\Output{
    $\mathcal{F}$ — 后端特征帧 $\{g_i \mapsto (\text{cam\_id}, [x, y, z, u, v, v_x, v_y]^\top)\}$\;
    $\mathcal{P}'$ — 更新后的全局特征绑定关系\;
    $\mathcal{G}'_{\text{grid}}$ — 更新后的空间哈希网格\;
    $n'_{\text{next}}$ — 更新后的全局ID计数器
}

\BlankLine
\tcp{阶段1：初始化当前帧状态}
$\mathcal{G}_{\text{cur}} \gets \varnothing$\tcp*{当前帧全局特征集合}
$\mathcal{G}_{\text{grid}} \gets \varnothing$\tcp*{当前帧空间哈希网格}

\BlankLine
\tcp{阶段2：传播上一帧已跟踪的全局特征}
\ForEach{$(g_i, \mathcal{B}_i) \in \mathcal{P}$}{
    \ForEach{$(ch_k, \text{lid}) \in \mathcal{B}_i$}{
        在通道 $ch_k$ 的局部ID列表中查找 $\text{lid}$\;
        \eIf{找到对应索引 $j$}{
            $gf \gets \mathcal{G}_{\text{cur}}[g_i]$\;
            $gf.\text{local\_ids}[ch_k] \gets \text{lid}$\;
            $gf.\text{pixel\_pts}[ch_k] \gets ch_k.\text{cur\_pts}[j]$\;
            $\textsc{InsertGrid}(\mathcal{G}_{\text{grid}}, g_i, ch_k.\text{cur\_pts}[j])$\;
        }{
            该通道绑定关系丢失，跳过\;
        }
    }
}

\BlankLine
\tcp{阶段3：注册未绑定的新局部特征}
\ForEach{$ch_k \in \mathcal{C}$}{
    \ForEach{$(\text{lid}_j, \mathbf{p}_j) \in ch_k$}{
        \If{$\textsc{IsBound}(\text{lid}_j, ch_k, \mathcal{G}_{\text{cur}})$}{
            \Continue\;
        }

        $\text{candidates} \gets \textsc{QueryNearby}(\mathcal{G}_{\text{grid}}, \mathbf{p}_j)$\;
        $\text{attached} \gets \text{false}$\;

        \ForEach{$g_i \in \text{candidates}$}{
            $gf \gets \mathcal{G}_{\text{cur}}[g_i]$\;
            \If{$gf.\text{local\_ids}$ 已含 $ch_k$ 或 $gf.\text{pixel\_pts} = \varnothing$}{
                \Continue\;
            }

            $d_{\min}^2 \gets \displaystyle\min_{(ch', \mathbf{p}') \in gf.\text{pixel\_pts}} \|\mathbf{p}_j - \mathbf{p}'\|^2$\;

            \If{$d_{\min}^2 \leq \tau_{\text{dist}}^2$}{
                $gf.\text{local\_ids}[ch_k] \gets \text{lid}_j$\;
                $gf.\text{pixel\_pts}[ch_k] \gets \mathbf{p}_j$\;
                $\textsc{InsertGrid}(\mathcal{G}_{\text{grid}}, g_i, \mathbf{p}_j)$\;
                $\text{attached} \gets \text{true}$\;
                \Break\;
            }
        }

        \If{not $\text{attached}$}{
            $g_{\text{new}} \gets n_{\text{next}}$\;
            $n_{\text{next}} \gets n_{\text{next}} + 1$\;
            $gf_{\text{new}}.\text{global\_id} \gets g_{\text{new}}$\;
            $gf_{\text{new}}.\text{local\_ids}[ch_k] \gets \text{lid}_j$\;
            $gf_{\text{new}}.\text{pixel\_pts}[ch_k] \gets \mathbf{p}_j$\;
            $\mathcal{G}_{\text{cur}}[g_{\text{new}}] \gets gf_{\text{new}}$\;
            $\textsc{InsertGrid}(\mathcal{G}_{\text{grid}}, g_{\text{new}}, \mathbf{p}_j)$\;
        }
    }
}

\BlankLine
\tcp{阶段4：构建后端特征帧输出}
$\mathcal{F} \gets \varnothing$\;
\ForEach{$(g_i, gf) \in \mathcal{G}_{\text{cur}}$}{
    \ForEach{$(ch_k, \text{lid}) \in gf.\text{local\_ids}$}{
        在通道 $ch_k$ 中查找 $\text{lid}$ 对应的索引 $j$\;
        \If{找到}{
            $\mathbf{x}_{\text{norm}} \gets ch_k.\text{cur\_un\_pts}[j]$\tcp*{归一化坐标}
            $\mathbf{x}_{\text{pix}} \gets ch_k.\text{cur\_pts}[j]$\tcp*{像素坐标}
            $\mathbf{v} \gets ch_k.\text{pts\_velocity}[j]$\tcp*{像素速度}
            $\mathcal{F}[g_i] \gets \bigl(0,\; [\mathbf{x}_{\text{norm}}.x,\; \mathbf{x}_{\text{norm}}.y,\; 1,\; \mathbf{x}_{\text{pix}}.x,\; \mathbf{x}_{\text{pix}}.y,\; \mathbf{v}.x,\; \mathbf{v}.y]^\top\bigr)$\;
            \Break\tcp*{每全局特征仅输出一个观测}
        }
    }
}

\BlankLine
\tcp{阶段5：持久化绑定关系供下一帧使用}
$\mathcal{P}' \gets \varnothing$\;
\ForEach{$(g_i, gf) \in \mathcal{G}_{\text{cur}}$}{
    \If{$gf.\text{local\_ids} \neq \varnothing$}{
        $\mathcal{P}'[g_i] \gets gf.\text{local\_ids}$\;
    }
}

\BlankLine
\Return $\mathcal{F},\; \mathcal{P}',\; \mathcal{G}_{\text{grid}},\; n_{\text{next}}$\;
\end{algorithm}

\vspace{1em}
\noindent\textbf{辅助函数：}

\begin{itemize}
    \item $\textsc{InsertGrid}(\mathcal{G}_{\text{grid}}, g_i, \mathbf{p})$：将全局ID $g_i$ 插入空间哈希网格，单元格坐标为 $(\lfloor p_x / s \rfloor, \lfloor p_y / s \rfloor)$，$s$ 为网格尺寸。
    \item $\textsc{QueryNearby}(\mathcal{G}_{\text{grid}}, \mathbf{p})$：查询 $\mathbf{p}$ 所在单元格及其8邻域单元格中的所有全局ID，返回候选列表。
    \item $\textsc{IsBound}(\text{lid}, ch, \mathcal{G}_{\text{cur}})$：检查局部ID $\text{lid}$ 在通道 $ch$ 中是否已被当前帧任一全局特征绑定。
\end{itemize}

\vspace{0.5em}
\noindent\textbf{数据结构：}

\begin{itemize}
    \item \textbf{GlobalFeature} $gf$：$\{$global\_id : int; local\_ids : map$\langle$channel\_name, local\_id$\rangle$; pixel\_pts : map$\langle$channel\_name, pixel\_coord$\rangle\}$
    \item \textbf{绑定关系} $\mathcal{P}$：map$\langle$global\_id, local\_ids$\rangle$，记录上一帧全局特征到各通道局部ID的映射
    \item \textbf{空间哈希} $\mathcal{G}_{\text{grid}}$：map$\langle$(cell\_x, cell\_y), [global\_id]$\rangle$，用于 $O(1)$ 邻域查询
\end{itemize}
